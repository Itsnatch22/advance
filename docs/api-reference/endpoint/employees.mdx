## List Employees

<ParamField query="organization_id" type="string" required>
  Organization identifier
</ParamField>

<ParamField query="page" type="integer" default="1">
  Page number for pagination
</ParamField>

<ParamField query="limit" type="integer" default="20">
  Number of results per page (max 100)
</ParamField>

### Request

<CodeGroup>

```bash cURL
curl https://api.eaziwage.com/v1/employees \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -G \
  -d organization_id=org_123 \
  -d page=1 \
  -d limit=20
```

```javascript JavaScript
const response = await fetch('https://api.eaziwage.com/v1/employees?organization_id=org_123', {
  headers: {
    'Authorization': 'Bearer YOUR_API_KEY'
  }
});
const data = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://api.eaziwage.com/v1/employees',
    headers={'Authorization': 'Bearer YOUR_API_KEY'},
    params={'organization_id': 'org_123'}
)
data = response.json()
```

</CodeGroup>

### Response

<ResponseField name="employees" type="array">
  Array of employee objects
  
  <Expandable title="properties">
    <ResponseField name="id" type="string">
      Unique employee identifier
    </ResponseField>
    <ResponseField name="first_name" type="string">
      Employee first name
    </ResponseField>
    <ResponseField name="last_name" type="string">
      Employee last name
    </ResponseField>
    <ResponseField name="email" type="string">
      Employee email address
    </ResponseField>
    <ResponseField name="phone_number" type="string">
      Employee phone number in E.164 format
    </ResponseField>
    <ResponseField name="status" type="string">
      Employee status: active, inactive, suspended
    </ResponseField>
    <ResponseField name="kyc_status" type="string">
      KYC verification status: pending, verified, rejected
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="pagination" type="object">
  <Expandable title="properties">
    <ResponseField name="total" type="integer">
      Total number of employees
    </ResponseField>
    <ResponseField name="page" type="integer">
      Current page number
    </ResponseField>
    <ResponseField name="limit" type="integer">
      Results per page
    </ResponseField>
  </Expandable>
</ResponseField>
